<body>
  <h1>Momentum Clicker</h1>
  <p>Momentum Points: <span id="points">0</span></p>
  <button id="clicker">Click for Momentum</button>

  <script>
    let points = parseInt(localStorage.getItem('momentumPoints')) || 0;
    const pointsDisplay = document.getElementById("points");
    const clickerButton = document.getElementById("clicker");

    function updateDisplay() {
      pointsDisplay.textContent = points;
      localStorage.setItem('momentumPoints', points);
    }

    clickerButton.addEventListener("click", () => {
      points += 1;
      updateDisplay();
    });

    updateDisplay();
  </script>
